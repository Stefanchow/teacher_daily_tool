  private sanitizeJsonString(str: string): string {
    return str
      // Aggressive sanitization: remove control characters but PRESERVE newlines/tabs for content
      // User's regex [\u0000-\u001F] includes \n \r \t. We must exclude them to preserve content formatting.
      // Range \u0000-\u0008 (NUL to BS)
      // \u0009 (TAB) - Keep
      // \u000A (LF) - Keep
      // \u000B (VT) - Remove
      // \u000C (FF) - Remove
      // \u000D (CR) - Keep
      // \u000E-\u001F (SO to US) - Remove
      .replace(/[\u0000-\u0008\u000B\u000C\u000E-\u001F\u007F-\u009F]/g, '')
      // Remove BOM
      .replace(/^\uFEFF/, '')
      // Remove trailing commas in arrays/objects
      .replace(/,(\s*[}\]])/g, '$1');
  }
